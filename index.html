The Task

from cormoran import *

class Task(Persistent):
        body = StringField()
        done = BooleanField(default=False)

Getting tasks

from tornado import web
from todo import models

class IndexHandler(web.RequestHandler):
        def initialize(self, store):
                self.store = store

        def get(self):
                self.render('index.html',
                        tasks=self.store.find(models.Task))

Adding tasks

import httplib

class IndexHandler(web.RequestHandler):
        def post(self):
                self.store.add(models.Task(
                        body=self.get_argument('body')))

                self.store.commit()
                self.set_status(httplib.CREATED)

Putting it all together

from tornado import web
from cormoran import *
from todo import handlers

def application(debug=False):
        sqlite = connect('sqlite:///tasks.sqlite')
        create_table(sqlite)

        return web.Application([
                (r'/', handlers.IndexHandler, dict(store=Store(sqlite)))
        ])
